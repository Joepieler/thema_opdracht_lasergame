251  void main() override {
252  enum states { WAIT_FOR_MESSAGE };
253  states state = states::WAIT_FOR_MESSAGE;
254  for(;;) {
255  switch( state ) {
256  case states::WAIT_FOR_MESSAGE: {
257  auto event = wait(
258  display_command_flag +
259  display_countdown_flag +
260  display_deaths_flag +
261  display_ammo_flag +
262  display_health_flag +
263  display_player_flag +
264  display_weapon_flag
265  );
266 
267  if( event == display_command_flag ) {
268  updateCommandValue( command_pool.read() );
269  }
270  else if( event == display_countdown_flag ) {
271  updateCountdownValue( countdown_pool.read() );
272  }
273  else if( event == display_deaths_flag ) {
274  hwlib::wait_ms( display_delay );
275  updateDeathsValue( deaths_pool.read() );
276  }
277  else if( event == display_ammo_flag) {
278  hwlib::wait_ms( display_delay );
279  updateAmmoValue( ammo_pool.read() );
280 
281  }
282  else if( event == display_health_flag ) {
283  hwlib::wait_ms( display_delay );
284  updateHealtValue( health_pool.read() );
285  }
286  else if( event == display_player_flag ) {
287  updatePlayerValue( player_pool.read() );
288  }
289  else if( event == display_weapon_flag ) {
290  updateWeaponValue( weapon_pool.read() );
291  }
292  break;
293  }
294  }
295  }
