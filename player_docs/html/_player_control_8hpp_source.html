139  void main() override {
140  enum states { INIT_GAME, PLAYING };
141  enum playing_states { ALIVE_ABLE_TO_SHOOT, ALIVE_NOT_ABLE_TO_SHOOT, DEAD };
142  states state = INIT_GAME;
143  playing_states playing_state;
144  ir_msg msg;
145  uint8_t game_length;
146 
147  for(;;) {
148  switch(state) {
149  case states::INIT_GAME: {
150  auto event = wait( player_number_channel + player_weapon_channel + msg_channel );
151  if( event == player_number_channel ) {
152  player_data.setPlayerID( player_number_channel.read() );
153  display_control.showPlayer( player_data.getPlayerID() );
154  }
155  else if( event == player_weapon_channel ) {
156  weapon.setWeapon( player_weapon_channel.read() );
157  display_control.showWeapon( weapon.getWeaponName(weapon.getWeaponID()) );
158  }
159  else if( event == msg_channel ){
160  msg = msg_channel.read();
161  if( msg.player == 0 && msg.data < 16 && msg.data > 0 ) {
162  game_length = msg.data;
163  buzzer_control.makeSound( buzzer_hit_length );
164  }
165  else if( msg.player == 0 && msg.data == 0) {
166  player_data.setHealth( player_data.getMaxHealth() );
167  weapon.setAmmo( weapon.getMaxAmmo() );
168  player_data.setDeaths( 0 );
169  buzzer_control.makeSound( buzzer_start_game_length );
170  hwlib::wait_ms( buzzer_start_game_length );
171  display_control.showHealth( player_data.getHealth() );
172  display_control.showAmmo( weapon.getAmmo() );
173  display_control.showDeaths( player_data.getDeaths() );
174  hwlib::wait_ms( initialization_pause );
175  for(int i = 5; i >= 0; i--){
176  display_control.showCountdown(i);
177  hwlib::wait_ms( countdown_interval );
178  }
179  trigger_flag.clear();
180  reload_flag.clear();
181  msg_channel.clear();
182  reload_led.set(0);
183  dead_led.set(0);
184  game_timer.set( ( game_length * 60000 ) * rtos::ms );
185  state = states::PLAYING;
186  playing_state = ALIVE_ABLE_TO_SHOOT;
187  }
188  }
189  break;
190  }
191 
192  case states::PLAYING: {
193  switch(playing_state){
194  case playing_states::ALIVE_ABLE_TO_SHOOT: {
195  auto event = wait( trigger_flag + reload_flag + msg_channel + game_timer );
196  if( event == game_timer ) {
197  state = states::INIT_GAME;
198  reload_led.set(1);
199  dead_led.set(1);
200  game_logs.printLogs();
201  game_logs.clearLogs();
202  }
203  else if( event == trigger_flag && weapon.getAmmo() > 0 ) {
204  shoot_control.shoot();
205  weapon.setAmmo( weapon.getAmmo() - 1 );
206  display_control.showAmmo( weapon.getAmmo() );
207  shot_delay_timer.set( (weapon.getShotDelay() * 100) * rtos::ms );
208  playing_state = playing_states::ALIVE_NOT_ABLE_TO_SHOOT;
209  }
210  else if( event == reload_flag ){
211  reload_timer.set( (weapon.getReloadTime() * 1000) * rtos::ms );
212  weapon.setAmmo( weapon.getMaxAmmo() );
213  reload_led.set( led_on );
214  playing_state = playing_states::ALIVE_NOT_ABLE_TO_SHOOT;
215  }
216  else if( event == msg_channel) {
217  msg = msg_channel.read();
218  if( msg.player > 0 && msg.player != player_data.getPlayerID() ) {
219  buzzer_control.makeSound( buzzer_hit_length );
220  game_logs.addLog( msg.player, weapon.getWeaponName(msg.data) );
221  if( weapon.getWeaponDamage(msg.data) >= player_data.getHealth() ) {
222  player_data.setHealth( 0 );
223  display_control.showHealth( player_data.getHealth() );
224  dead_led.set( led_on );
225  playing_state = playing_states::DEAD;
226  }
227  else{
228  player_data.setHealth( (player_data.getHealth() - weapon.getWeaponDamage( msg.data ) ) );
229  display_control.showHealth( player_data.getHealth() );
230  }
231  }
232  }
233  break;
234  }
235 
236  case playing_states::ALIVE_NOT_ABLE_TO_SHOOT: {
237  auto event = wait( shot_delay_timer + reload_timer + msg_channel + game_timer );
238  if( event == game_timer ){
239  state = states::INIT_GAME;
240  reload_led.set(1);
241  dead_led.set(1);
242  game_logs.printLogs();
243  game_logs.clearLogs();
244  }
245  else if( event == msg_channel ) {
246  msg = msg_channel.read();
247  if( msg.player > 0 && msg.player != player_data.getPlayerID() ) {
248  buzzer_control.makeSound( buzzer_hit_length );
249  game_logs.addLog( msg.player, weapon.getWeaponName(msg.data) );
250  if( weapon.getWeaponDamage(msg.data) >= player_data.getHealth() ) {
251  player_data.setHealth( 0 );
252  display_control.showHealth( player_data.getHealth() );
253  dead_led.set( led_on );
254  playing_state = playing_states::DEAD;
255  }
256  else{
257  player_data.setHealth( (player_data.getHealth() - weapon.getWeaponDamage(msg.data) ) );
258  display_control.showHealth( player_data.getHealth() );
259  }
260  }
261  }
262  else if( event == shot_delay_timer ) {
263  trigger_flag.clear();
264  playing_state = playing_states::ALIVE_ABLE_TO_SHOOT;
265  }
266  else if( event == reload_timer ) {
267  reload_flag.clear();
268  display_control.showAmmo( weapon.getAmmo() );
269  reload_led.set( led_off );
270  playing_state = playing_states::ALIVE_ABLE_TO_SHOOT;
271  }
272  break;
273  }
274 
275  case playing_states::DEAD: {
276  reload_led.set(0);
277  death_timer.set( (player_data.getDeathLength() * 1000) * rtos::ms );
278  player_data.setDeaths( (player_data.getDeaths() + 1 ) );
279  display_control.showDeaths( player_data.getDeaths() );
280  auto event = wait( death_timer + game_timer );
281  if( event == game_timer ) {
282  state = states::INIT_GAME;
283  reload_led.set(1);
284  dead_led.set(1);
285  game_logs.printLogs();
286  game_logs.clearLogs();
287  }
288  else if ( event == death_timer ) {
289  reload_flag.clear();
290  trigger_flag.clear();
291  player_data.setHealth( player_data.getMaxHealth() );
292  weapon.setAmmo( weapon.getMaxAmmo() );
293  display_control.showHealth( player_data.getHealth() );
294  display_control.showAmmo( weapon.getAmmo() );
295  msg_channel.clear();
296  dead_led.set( led_off );
297  playing_state = playing_states::ALIVE_ABLE_TO_SHOOT;
298  }
299  }
300  break;
301  }
302  break;
303  }
304  }
305  }
306  }
