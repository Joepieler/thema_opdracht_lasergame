64  void main() override {
65  enum status { IDLE, TYPE_COMMAND, SEND_COMMAND };
66  status state = status::IDLE;
67  int command = 0;
68  for (;;) {
69  switch( state ) {
70  case status::IDLE: {
71  wait( receive_clock );
72  char keyvalue = keypad.getc();
73  if( keyvalue == 'C' ) {
74  state = status::TYPE_COMMAND;
75  }
76  break;
77  }
78  case status::TYPE_COMMAND: {
79  wait( receive_clock );
80  char keyvalue = keypad.getc();
81  if( keyvalue == 'C' ){
82  command = 0;
83  display_control.clearCommandValue();
84  state = status::IDLE;
85  }
86  else if( keyvalue > '0' && keyvalue <='9' ) {
87  command *= 10;
88  command += getNumber( keyvalue );
89  }
90  else if( keyvalue == '*' ) {
91  game_leader.startGame();
92  command = '*';
93  display_control.showCommand( command );
94  }
95  else if( keyvalue == '#' ){
96  if( command > 0 && command <= max_command ) {
97  display_control.showCommand( command );
98  state = status::SEND_COMMAND;
99  }
100  else {
101  command = 0;
102  }
103  }
104  break;
105  }
106  case status::SEND_COMMAND: {
107  wait( receive_clock );
108  char keyvalue = keypad.getc();
109  if( keyvalue == '#' ) {
110  game_leader.setGameLength( command );
111  }
112  else if( keyvalue == 'C' ) {
113  command = 0;
114  display_control.clearCommandValue();
115  state = status::IDLE;
116  }
